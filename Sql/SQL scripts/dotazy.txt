SELECT * FROM ADRESY WHERE (prodejny_idprodejny is null or IDADRESY = (select MAX(IDADRESY) FROM  ADRESY WHERE PRODEJNY_IDPRODEJNY = 1)) and zamestnanci_idzamestnance is null;

SELECT z.IDZAMESTNANCE, z.jmeno, z.prijmeni, z.telefonniCislo, z.rodneCislo, z.ZAMESTNANCI_IDZAMESTNANCE, poz.nazev
FROM ZAMESTNANCI z
join pozice poz on poz.idpozice = z.pozice_idpozice
START WITH z.ZAMESTNANCI_IDZAMESTNANCE IS NULL
CONNECT BY PRIOR z.IDZAMESTNANCE = z.ZAMESTNANCI_IDZAMESTNANCE order by z.ZAMESTNANCI_IDZAMESTNANCE desc;
